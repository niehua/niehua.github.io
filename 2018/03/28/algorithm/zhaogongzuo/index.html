找最佳匹配度工作

Posted on 2018-03-28 | In Java | Visitors:
2018年网易本科实习生上机考试题

题目内容

为了找到自己满足的工作，牛牛收集了每种工作的难度和报酬，
选择工作的标准时难度不超过自身的能力值的情况下选择报酬最高的工作，
当小伙伴很多时，这个任务交给了我。。。！

  • 思路

/*
开始理解错误，认为工作难度适合报酬成正比，然而实际情况是难度低的，也可能工资更高
先对工作和能力排序，当找到一个匹配度合适的伙伴，下次寻找时，只需要
在难度更高里面找

N， M都达到了$10^5$，很明显，对于每次询问要采用$O(logn)$或$O(1)$的算法，$O(logn)$可以二分查找，
$O(1)$可以通过预处理来达到目标。

因此，先按每个工作的能力值升序排序，这样子可以二分查找到每个同学能做的所有工作，
然后再这些工作中找到薪酬最大的；
如何快速找薪酬最大的，这里就要对刚刚升序排序的输出进行区间最大值的预处理了，
递推一下就可以得到[1, i]区间的最大值，
找区间薪酬最大，只需要访问下dp[i]就行了。

n:工作种类
m:小伙伴人数
d:难度
p:报酬
a:能力值
*/

代码

  • 错误版本
import java.util.Scanner;

public class T {

    static void def(int n,int m,int[] d,int[] p,int[] a)
    {
        sort(d,n);
        sort(p,n);
        sort(a,m);

        int i=0;
        for(int j=0;j<m;j++)//一共有M个人，得找m次
        {
            //当难度小于能力时，循环增加难度，找到最贴近能力的难度，此时报酬最大化
            while(i<n&&d[i]<=a[j])  //a[j]对应此人的能力，定值-难点：等号是否取
            {
                i++;
            }
            i=i-1;//下次寻找从难度di找，由于排好序的，后面的人能力肯定更强
            System.out.println(p[i]);
        }
    }
    static void sort(int[] len,int n){
        int temp=0;
        for(int i=0;i<n-1;i++){
            for(int j=0;j<n-i-1;j++){
                if(len[j]>len[j+1]){
                    temp=len[j];
                    len[j]=len[j+1];
                    len[j+1]=temp;
                }
            }
        }
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int m = sc.nextInt();
        int[] d=new int[n];
        int[] p=new int[n];
        int[] a=new int[n];
        int i;
        for(i=0;i<n;i++)
        {
            d[i]=sc.nextInt();
            p[i]=sc.nextInt();;
        }
        for(i=0;i<m;i++)
        {
            a[i]=sc.nextInt();
        }
        def(n,m,d,p,a);
    }
}

  • 正确版本

    import java.util.*;

    class Job {

    public int d;
public int p;

    Job(int d, int p)
    {
        this.d = d;
        this.p = p;
    }
}

public class T 
{

    //  begin  0
    //    right    n
    //  p    报酬
    //一般查找，时间效率不够
    public static int half(Job[] jarr, int n, int p) 
    {
        int i=0;
        for( ;i<n&&jarr[i].d <= p;i++)
        {
        }
        return i-1;
    }

    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        while (sc.hasNext()) 
        {
            int n = sc.nextInt(), m = sc.nextInt();
            Job[] jarr = new Job[n];
            for (int i = 0; i < n; i++)
            {
                jarr[i] = new Job(sc.nextInt(), sc.nextInt());
            }
            Arrays.sort(jarr, new Comparator<Job>() //匿名内部类
            {
                @Override
                public int compare(Job j1, Job j2) {
                    return j1.d - j2.d;
                }
            });
            int[] _p = new int[n];
            _p[0] = jarr[0].p;
            //工作难度是唯一的，递推提取报酬到新数组,更改报酬，保证区间报酬最大，如下
            //200  400
            //210  420
            //230  300 -> 420
            //240  500
            //250  400 -> 500
            //260  390 -> 500
            for(int i=1;i<n;i++)
            {
                _p[i]=Math.max(jarr[i].p,_p[i-1]);
            }
            while (m-->0)
            {
                int index=half(jarr,n,sc.nextInt());
                System.out.println(index==-1 ? 0 : _p[index] );
            }
        }
    }
}
