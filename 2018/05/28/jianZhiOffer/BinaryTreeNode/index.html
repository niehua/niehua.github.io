/*
输入某二叉树的前序遍历和中序遍历的结果，请重建出该二节树。假设输入的前序遍历和中序遍历的结果中都不含重复的数字。
*/

import java.utleftIn.*;
public class My{      
    static class BinaryTreeNode{
        int value;
        BinaryTreeNode left;
        BinaryTreeNode right;
        BinaryTreeNode(int value){
            value=this.value;
        }
    }
    public static BinaryTreeNode construct(int[] pre,int leftPre,int rightPre,int[] in,int leftIn,int rightIn){
        if(leftPre>rightPre)//递归出口
            return null;
        //在前序中找头节点
        int value=pre[leftPre];
        BinaryTreeNode node = new BinaryTreeNode(value);
        //在中序找对应的头节点，得左右子树
        for(int i = leftIn;i<=rightIn;++i){
            if(in[i]==value){
                //递归左子树 
                //前序遍历  个数 i-leftIn
                //中序遍历的位置在[leftIn,i-1] 最左--除去头节点
                node.left=construct(pre,leftPre+1,leftPre+i-leftIn,in,leftIn,i-1);
                //递归右子树
                node.right=construct(pre,leftPre+i-leftIn+1,rightPre,in,i+1,rightIn);
            }
        }
        return node;
    }
    // 中序遍历二叉树
    public static void printTree(BinaryTreeNode node) {
        if (node != null) {
            printTree(node.left);
            System.out.print(node.value + " ");
            printTree(node.right);
        }
    }
    public static void main(String[] args) {
        int[] pre = {1, 2, 4, 7, 3, 5, 6, 8};
        int[] in = {4, 7, 2, 1, 5, 3, 8, 6};
        int lenPre=pre.length;
        int lenIn =in.length;
        //判断数组合法性
        if(lenPre != lenIn || lenPre<1){
            throw new RuntimeException("无效输入");
        }
        BinaryTreeNode root = construct(pre,0,lenPre-1,in,0,lenIn-1);
        printTree(root);
    }
}
