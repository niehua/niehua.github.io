Huffman树构造算法

原理

哈夫曼提出构造最优前缀码的贪心算法，由此产生的编码方案称为哈夫曼编码。其构造步骤如下：
(1)哈夫曼算法以自底向上的方式构造表示最优前缀码的二叉树T。
(2)算法以|C|个叶结点开始，执行|C|－1次的“合并”运算后产生最终所要求的树T。
(3)假设编码字符集中每一字符c的频率是f(c)。以f为键值的优先队列Q用在贪心选择时有效地确定算法
当前要合并的2棵具有最小频率的树。一旦2棵具有最小频率的树合并后，产生一棵新的树，
其频率为合并的2棵树的频率之和，并将新树插入优先队列Q。经过n－1次的合并后，优先队列中只剩下一棵树，
即所要求的树T。

源代码：

import java.util.*;
//概率 
abstract class HuffmanTree implements Comparable<HuffmanTree> {
    public final double p; 
    public HuffmanTree(double p) { this.p = p; }
    @Override
    public int compareTo(HuffmanTree tree) {
    //return this.p - tree.p;
        return new Double(this.p).compareTo(new Double(tree.p));
    }
}
 //信源符号和概率
class HuffmanLeaf extends HuffmanTree {
    public final String s; 
    public HuffmanLeaf(String s,double p) {
    super(p);
    this.s = s;
    }
}
//左右节点
class HuffmanNode extends HuffmanTree {
    public final HuffmanTree left, right; 
    public HuffmanNode(HuffmanTree l, HuffmanTree r) {
    super(l.p + r.p);
    left = l;
    right = r;
    }
}
public class Huffman {
    private static int q = 0;
    //递归回溯编码
    public static void printCodes(HuffmanTree tree,StringBuilder prefix) {
    assert tree != null;
    if (tree instanceof HuffmanLeaf) {
        HuffmanLeaf ht = (HuffmanLeaf)tree;
        System.out.println(ht.s + "\t->\t" + prefix);
    } else if (tree instanceof HuffmanNode) {
        HuffmanNode node = (HuffmanNode)tree;
        //遍历左节点
        prefix.append('0');
        printCodes(node.left, prefix);
            //deleteCharAt()移除此序列指定位置上的 char。此序列将缩短一个 char。
        prefix.deleteCharAt(prefix.length()-1);//在遍历右节点之前，删除刚遍历的左节点
            //遍历右节点
        prefix.append('1');
        printCodes(node.right, prefix);
            prefix.deleteCharAt(prefix.length()-1);//在遍历左节点之前，删除刚遍历的右节点
    }
    }
    public static void main(String[] args) {
        //放到优先级队列中，优先级队列的元素按照其自然顺序进行排序，不允许插入不可比较的对象，
        //队列的头是按指定排序方式确定的最小元素
        PriorityQueue<HuffmanTree> trees = new PriorityQueue<HuffmanTree>();
        Scanner sc = new Scanner(System.in);
        System.out.println("请输入符号个数");
        //while (sc.hasNext()){
            q = sc.nextInt();//信源符号个数
            HuffmanLeaf[] ht = new HuffmanLeaf[q];
            //信源输入
            System.out.println("请分别输入信源与概率");
            for(int i = 0;i<q;i++){
                ht[i]=new HuffmanLeaf(sc.next(),sc.nextDouble());
                trees.offer(ht[i]);//添加到队列中
            }
            /*System.out.println("\n********队列表********");
            for(int i = 0;i<q;i++){
                System.out.println(ht[i].s + "\t->\t" + ht[i].p);
            }*/
            //只有一或两个字符直接返回
            if(q==1){
                System.out.println("\n********编码表********");
                System.out.println(ht[0].s+ "\t->\t0");    
            }
            if(q==2){
                System.out.println("\n********编码表********");
                System.out.println(ht[0].s+ "\t->\t0");    
                System.out.println(ht[1].s+ "\t->\t1");    
            }
            assert trees.size() > 0;//队列不为空
            //构建huffuman树，每次弹出两个对象，直到只剩一个根节点
            //进入队列若相等，则拍到相等数后边，作为第二个节点右节点取出
            while (trees.size() > 1) {
                HuffmanTree a = trees.poll();
                HuffmanTree b = trees.poll();
                trees.offer(new HuffmanNode(a, b));
            }
            HuffmanTree tree = trees.poll();//弹出根节点
            System.out.println("\n********编码表********");
            //打印编码
            printCodes(tree,new StringBuilder());
        //}
    }
}

总结

用到的主要数据结构：PriorityQueue<>，自定义构建的二叉树，其中构建的信源类，实现了接口Comparable，并覆写了compareTo()方法，具有比较性
放到优先级队列中，优先级队列的元素按照其自然顺序进行排序，不允许插入不可比较的对象，队列的头是按指定排序方式确定的最小元素
利用HuffmanTree的两个子类，一个位信源类，一个位节点类，把添加到队列的信源以及后面进入队列的节点一次弹出，知道只剩下定点（root），在弹出顶点给输出函数，这个函数是一个递归回溯的函数，自上而下遍历每一个叶节点，并输出编码
