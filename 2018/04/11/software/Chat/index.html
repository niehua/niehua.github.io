局域网聊天

Posted on 2018-04-11 | In Java | Visitors:
在同一个局域网内的人可以聊天，默认学校的广播地址（10.18.141.255）是发送消息给所有人，可以自己设置需要聊天的目标地址指定目标

环境

eclipce + java1.8

发送者

package niehua.send;
import java.net.DatagramPacket;
import java.net.DatagramSocket;
import java.net.InetAddress;
import javax.swing.JOptionPane;
import niehua.windows.ChatWindows;

public class UdpS implements Runnable{
    private int portDes = 10000;//目的主机端口
    private int portSrc = 8000;//本机发送端口
    private String ms =  null;//发送的消息
    private DatagramSocket ds =null;
    public  static String addr = "10.18.141.255";//目的IP地址

    public UdpS(String ms) {
        this.ms = ms +"\r\n";
    }
    public  void sendMs() {
        try{
            ds = new DatagramSocket(portSrc);
            byte[] buf=ms.getBytes();
            DatagramPacket dp=new DatagramPacket(buf,buf.length,InetAddress.getByName( addr ),portDes);//在给定主机名的情况下确定目的主机的IP地址
            //构造数据报包，用来将长度为 length 的包发送到指定主机上的指定端口号。
            ds.send(dp);//发送
        }
        catch(Exception e1){
            JOptionPane.showMessageDialog(ChatWindows.contentPane,"可能输入ip有错或不可达，请重新输入", "warning", JOptionPane.ERROR_MESSAGE);
        }
        finally{
            try{  ds.close();}
            catch(Exception e2){  throw new  RuntimeException("关闭发送套接字异常"); }
        }
    }
    public static String getIp(){
        try {
            return InetAddress.getLocalHost().getHostAddress()+" : ";
        }catch(Exception e) { throw new RuntimeException("获取发送者ip失败");}
    }
    public void run(){
            sendMs();
            ChatWindows.txtrDasa.append("                    from "+ getIp()+"\r\n"+this.ms);
            ChatWindows.textArea.setText("");
    }

}

接收者

package niehua.rece;
import java.net.DatagramPacket;
import java.net.DatagramSocket;
import niehua.send.UdpS;
import niehua.windows.ChatWindows;

public class UdpR implements Runnable
{
    public   String ip = null;
    private DatagramSocket ds =null;
    private int portSrc = 10000;//本主机端口
    public  String ms = null;
    public  UdpR() { }
    public  void recMs() 
    {    
        try {//怎样使发送消息时，接收程序开着，没消息是关着，只有一开窗口就执行接收程序。
            ds = new DatagramSocket(portSrc);//只绑定一次端口，开始写在while里面，循环绑定错误
            while(true){
                byte[] buf=new byte[1024];//buf.length=1024，且默认位0；
                DatagramPacket dp=new DatagramPacket(buf,buf.length);//buf.length=接收到的数据长度，一般不是1024
                ds.receive(dp);//接受,同样时阻塞方法
                ip=dp.getAddress().getHostAddress()+" : ";//返回某台机器的 IP 地址(InetAddress)，此数据报将要发往该机器或者是从该机器接收到的。
                ms =new String(dp.getData(),0,dp.getLength());//返回数据缓冲区。通过使用平台的默认字符集解码指定的 byte 子数组，构造一个新的 String。
                int port =dp.getPort();
                if ( !ip.equals(UdpS.getIp())){
                    ChatWindows.txtrDasa.append("from "+ip+port+"\r\n"+this.ms);
                }
                //System.out.println("from "+ip+port+"\r\n"+ms);
            }
        }
        catch(Exception e) {
            //throw new  RuntimeException("接收异常");
            System.out.println(e.toString());
        }
        finally{
            try{  ds.close();}
            catch(Exception e2){  throw new  RuntimeException("关闭接收套接字异常"); }
        }
    }
    public void run(){
            recMs();
    }    
}

package niehua.windows;

import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import javax.swing.JFrame;
import javax.swing.JLabel;
import javax.swing.JOptionPane;
import javax.swing.JPanel;
import javax.swing.JScrollPane;
import javax.swing.JTextArea;
import javax.swing.JTextField;
import javax.swing.ScrollPaneConstants;
import javax.swing.border.EmptyBorder;

import niehua.rece.UdpR;
import niehua.send.UdpS;

public class ChatWindows extends JFrame {
    //这个是为了标识当前的版本。主要用于用于远程调用。RMI这类的场景。
    private static final long serialVersionUID = 1L;
    public  static JPanel contentPane;
    public  static TextArea textArea = null;
    public  static JTextArea txtrDasa = null;
    private JTextField txtD;
    public static void main(String[] args) {
        EventQueue.invokeLater(new Runnable() {
            public void run() {
                try {
                    ChatWindows frame = new ChatWindows();
                    frame.setVisible(true);
                } catch (Exception e) {
                    e.printStackTrace();
                }
            }
        });
    }

    public ChatWindows() {
        setTitle("Chat Happy\r\n");
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setBounds(100, 100, 1207, 720);
        contentPane = new JPanel();
        contentPane.setBorder(new EmptyBorder(5, 5, 5, 5));
        setContentPane(contentPane);
        contentPane.setLayout(null);

        init();

        textArea = new TextArea();//发送文本域
        textArea.setBounds(591, 281, 584, 326);
        textArea.setText("");
        textArea.setFont(new Font(Font.DIALOG_INPUT,Font.BOLD,24));
        contentPane.add(textArea);

        JScrollPane scrollPane = new JScrollPane();
        scrollPane.setHorizontalScrollBarPolicy(ScrollPaneConstants.HORIZONTAL_SCROLLBAR_NEVER);
        scrollPane.setBounds(0, 15, 584, 634);
        contentPane.add(scrollPane);

        txtrDasa = new JTextArea();//聊天记录
        txtrDasa.setEditable(false);
        txtrDasa.setLineWrap(true);
        txtrDasa.setText("");
        txtrDasa.setFont(new Font(Font.DIALOG_INPUT,Font.BOLD,24));
        scrollPane.setViewportView(txtrDasa);

        JLabel lblChatRecord = new JLabel("                                         Chat record");
        scrollPane.setColumnHeaderView(lblChatRecord);

        Button button = new Button("Send");
        button.setBackground(Color.LIGHT_GRAY);
        button.setForeground(Color.BLACK);
        button.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent arg0) {
                String ms = textArea.getText();
                if(!ms.isEmpty()) {
                    new Thread(new UdpS(ms)).start(); 
                }
            }
        });
        button.setBounds(591, 613, 584, 36);
        contentPane.add(button);

        JLabel lblTo = new JLabel("Please fill in IP where you want to send it to");
        lblTo.setFont(new Font("宋体", Font.PLAIN, 24));
        lblTo.setBounds(591, 14, 584, 21);
        contentPane.add(lblTo);

        txtD = new JTextField();
        txtD.setFont(new Font(Font.DIALOG_INPUT,Font.BOLD,24));
        txtD.setText("");
        txtD.setBounds(591, 50, 584, 57);
        contentPane.add(txtD);
        txtD.setColumns(10);

        Button button_1 = new Button("Confirm");
        button_1.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent arg0) {
                String ip = txtD.getText();
                if(ip.length()>=11) {
                    UdpS.addr = ip; 
                }
                else{
                    JOptionPane.showMessageDialog(contentPane,"可能输入ip有错或不可达，请重新输入", "warning", JOptionPane.ERROR_MESSAGE);
                }
            }
        });
        button_1.setBackground(Color.LIGHT_GRAY);
        button_1.setActionCommand("Confirm");
        button_1.setForeground(Color.BLACK);
        button_1.setBounds(591, 128, 592, 30);
        contentPane.add(button_1);
    }
    public void init()
    {
        new Thread(new UdpR()).start(); 
    }
}
