public class CapturePacket implements Runnable{


    public  static int devsId = 0;
    public  static String str=null;
    private static String user="41524211";
    /*回掉模式模式捕获数据包，我们创建一个包处理程序来从libpcap loop中接收数据包*/
    public static void packetHandler(Pcap pcap)
   {

    PcapPacketHandler<String> jpacketHandler = new PcapPacketHandler<String>()
    {  
        int order =1;
            @Override
        public void nextPacket(PcapPacket packet, String user)//阻塞方法，每次loop从这里开始
            {   
                //%-4d，仅使用与数据规模在四位数以下的情况，超出则会左对齐出错。
//                System.out.printf("\nReceived  packet at %s caplen=%-4d wirelen=%-4d %s  order: %d\n",  
//                    new Date(packet.getCaptureHeader().timestampInMillis()),   
//                    packet.getCaptureHeader().caplen(),              // 实际捕获的长度
//                    packet.getCaptureHeader().wirelen(),             // 原来长度
//                    user,                                            // 用户信息   
//                    order++); 
//                //System.out.println(packet);
                PacketMatch pm=PacketMatch.getPm();
                pm.handlePacket(packet,CapturePacket.user,order++);//由于上面覆盖的的nextPacket()方法中也有user,所以需指定
         }  
        };  

/*
第四，我们进入循环并告诉它捕获10个数据包。 循环方法执行pcap.datalink（）DLT值到JProtocol ID的映射
JScanner需要它。 扫描器扫描数据包缓冲区并进行解析头。这个映射是自动完成的，
尽管存在循环方法的变体，允许程序员准确地指该协议ID用作此pcap接口的数据链路类型。
*/  
    pcap.loop(1, jpacketHandler, user);  
        //最后一定要关闭pcap，否则抛出异常
    pcap.close();  
   }  
    public void run(){    
        PacketMatch.buf = new StringBuilder();//清空缓冲区
        CapturePacket.packetHandler(PacketMethod.open(devsId ));
        PacketMethod.fileWriter(PacketMatch.buf.toString());//写入日志文件
        MainWindows.txtrArea.setText(new String(PacketMatch.buf));//写入显示界面
    }
}

package niehua.action;

import java.util.Date;

import org.jnetpcap.packet.PcapPacket;

import org.jnetpcap.protocol.tcpip.Udp;  
public class PacketMatch 
{
    private Arp arp = new Arp();  
    private Ip4 ip  = new Ip4();  
    private Tcp tcp = new Tcp();  
    private Udp udp = new Udp(); 
    private Ethernet eth = new Ethernet(); 

    public  static StringBuilder bufEth = new StringBuilder();
    public  static StringBuilder bufArp = new StringBuilder();
    public  static StringBuilder bufIp  = new StringBuilder();
    public  static StringBuilder bufTcp = new StringBuilder();
    public  static StringBuilder bufUdp = new StringBuilder();

    private static String str=null;//单个协议的数据包
    public  static StringBuilder buf= new StringBuilder();//所有协议的数据包，传给输出界面以及日志文件


    private static PacketMatch pm=new PacketMatch();  
    private PacketMatch() {}
    public static PacketMatch getPm()
    {
        return pm;
    }
    public void handlePacket(PcapPacket packet,String user,int order){//根据包头选择不同的规则 
        str = "\r\nReceived  packet at " + new Date(packet.getCaptureHeader().timestampInMillis())
                + "  caplen="+packet.getCaptureHeader().caplen()
                + "  wirelen="+ packet.getCaptureHeader().wirelen()
                + "  "+user
                + "  order: "+order;

        buf.append(str + "\r\n");
    System.out.println(str);

    //hasHeader(int id)检查在解码的数据包内是否存在具有指定数字ID的所请求的头的实例。
    if (packet.hasHeader(eth)) {  
        handleEth(packet);  
        } 
    if (packet.hasHeader(arp)) {  
        handleArp(packet);  
        } 
        if (packet.hasHeader(ip)) {  
        handleIp(packet);  
        }   
        if (packet.hasHeader(tcp)) {  
        handleTcp(packet);  
        }  
        if (packet.hasHeader(udp)) {  
        handleUdp(packet);  
        } 


    }public void handleEth(PcapPacket packet)
    {
        //packet.getHeader(arp);  
        str= eth.toString();
        bufEth.append(str+ "\r\n");
        buf.append(str + "\r\n");
    System.out.println(str);
    }
    public void handleArp(PcapPacket packet)
    {
        //packet.getHeader(arp);  
        str= arp.toString();
        bufArp.append(str + "\r\n");
        buf.append(str + "\r\n");
    System.out.println(str);
    }
    public void handleIp(PcapPacket packet)
    {
        //packet.getHeader(ip);  
    byte[] sIP = new byte[4], dIP = new byte[4];  
    sIP = ip.source();  //返回的是对象引用 eg:[B@d041cf 
    dIP = ip.destination();  
    String srcIP = org.jnetpcap.packet.format.FormatUtils.ip(sIP);  
    String dstIP = org.jnetpcap.packet.format.FormatUtils.ip(dIP); 
    str= "源IP:"+srcIP+"       目的IP:" +dstIP;
    bufIp.append(str + "\r\n");
    buf.append(str + "\r\n");
    System.out.println(str);
    }
    public void handleTcp(PcapPacket packet)
    {
        //packet.getHeader(tcp);
        str= "Tcp源端口:"+tcp.source()+"  Tcp目的端口:" +tcp.destination();
        bufTcp.append(str + "\r\n");
        buf.append(str + "\r\n");
        System.out.println(str);
    }

package niehua.action;

import org.jnetpcap.Pcap;
import org.jnetpcap.packet.JMemoryPacket;
import org.jnetpcap.packet.JPacket;
import org.jnetpcap.protocol.JProtocol;
import org.jnetpcap.protocol.lan.Ethernet;
import org.jnetpcap.protocol.network.Ip4;
import org.jnetpcap.protocol.tcpip.Tcp;

import niehua.util.PacketMethod;

public class SendPacket implements Runnable{
    public  static int devsId = 0;
    private  void sendPacket(Pcap pcap)
    {      
        JPacket packet =  
                new JMemoryPacket(JProtocol.ETHERNET_ID,  
              " 001801bf 6adc0025 4bb7afec"+"0800" + "4500"   //MAC目的，源地址 14=12+2
            + " 0041a983 40004006 d69a"+"c0a8 00342f8c"        //IP源，目的地址20=2+10+8
            + " ca30" + "c3ef 008f"+"2e80 11f52ea8 4b578018 " //TCP源，目的端口号32=4+10+16+2
            + " ffffa6ea 00000101 080a152e ef03002a "  
            + " 2c94" + "3538 322e3430 204e4f4f 500d0a");  //79字节  79=14+20+32+13

            Ip4 ip = packet.getHeader(new Ip4());  
            Tcp tcp = packet.getHeader(new Tcp());  

            tcp.destination(80);  
            System.out.println(tcp.destination());//80

            ip.checksum(ip.calculateChecksum());  
            tcp.checksum(tcp.calculateChecksum());

            packet.scan(Ethernet.ID);  

            if (pcap.sendPacket(packet) != Pcap.OK) {  
              System.err.println(pcap.getErr());  
            } 

            pcap.close();    
    }  
    public void run(){    
        System.out.println(devsId);
        sendPacket(PacketMethod.open(devsId ));
    }

}

//Offline Cpature
/**PcapPacket
*一个pcap数据包。 完全解码的数据包，提供对解码过程中确定的协议头的
*访问。 PcapPacket类旨在与pcap库一起使用。 它不能用于从仅包含数据包
*数据的外部内存缓冲区创建新数据包，例如准备从网络接口发送的数据包。
*您可以使用JMemoryPacket从头开始创建内存中的数据包。
*PcapPackets需要在捕获数据包时由libpcap提供的PcapHeader。
*此外，PcapPacket还包含已解码的状态信息，可以使用友好的Java API
*和编译时类型安全性来查询数据包的内容。
*/

/**Pcap
* Pcap这个类是一个和 libpcap和winpcap 中 pcap_t 结构等同的主要类。它用于提供一个各种
* 库方法到java 的直接映射
* Pcap提供几个静态方法，用来发现网络接口，随后打开 openlive，openDead，或者 
* openOffline pcap捕获会话。以上都是返回一个 Pcap 对象。这个对象是C语言的pcap_t结构
* java vm 地址空间的支持。任何在Pcap对象上 的 非静态操作，都会转换为使用 JNI api 匹配 
* Libpcap 中 pcap_t C 的方法来完成调用。
* 用了以上提及的3个静态方法，一定要调用 close（） 方法来释放 Libpcap 的资源
* 如果 Pcap已经关闭了，但是非静态方法还是调用了，那就会抛出一个 IllegalStateException
*/


/** 通过Pcap.findAllDevs(alldevs, errbuf) 这个静态方法
* 将所有本机网卡加入到alldevs的List<PcapIf >中。然后用户可以选择
* 一个网卡进行监听；
* 这个方法构造了可以用pcap_open_live()打开的所有网络设备
* 这个列表中的元素都是 pcap_if_t，
* name 一个指向设备名字的指针；
* adderess 是一个接口的地址列表的第一个元素的指针；
* flag 一个PCAP_IF_LOOPBACK标记接口是否是loopback的
* 失败返回-1，成功返回0
*/ 

/**openlive方法：这个方法打开一个和指定网络设备有关的，活跃的捕获器 
* 参数：snaplen指定的是可以捕获的最大的byte数，
* 如果snaplen的值比我们捕获的包的大小要小的话，
* 那么只有snaplen大小的数据会被捕获并以packet data的形式提供。
* IP协议用16位来表示IP的数据包长度，所有最大长度是65535的长度
* 这个长度对于大多数的网络是足够捕获全部的数据包的
* 参数：flags promisc指定了接口是promisc模式的，也就是混杂模式，
* 混杂模式是网卡几种工作模式之一，比较于直接模式：
* 直接模式只接收mac地址是自己的帧，
* 但是混杂模式是让网卡接收所有的流过网卡的帧，达到了网络信息监视捕捉的目的
* 参数：timeout 这个参数使得捕获报后等待一定的时间，来捕获更多的数据包，
* 然后一次操作读多个包，不过不是所有的平台都支持，不支持的会自动忽略这个参数
* 参数：errbuf pcap_open_live()失败返回NULL的错误信息，或者成功时候的警告信息 
*/
/**捕捉到数据包后当然要进行分析。在这里我们使用继承JPacketHandler来实现自己的处理方法。
* 在JPacketHandler有 一个nextPacket(JPacket packet, T user) 方法，
* 这是典型的通过事件机制来实现处理数据包的方法。每当Pcap嗅探到一个数据包后，
* 他就会调用用户之前绑定的分析器中的nextPacket方法进 行处理。注意这个方法是阻塞的，
* 也就避免了潜在的同步问题。传进的JPacket参数包含了这个数据包中的所有信息，
* 通过不同的内置Header分析器可 以分析不同的协议。
*/

package niehua.util;

import java.io.FileWriter;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;

import org.jnetpcap.Pcap;
import org.jnetpcap.PcapIf;  

//用来完成数据包的封装，发送的工具类以及设备的例表输出。
public class PacketMethod 
{
    private static StringBuilder errbuf = new StringBuilder();
    private static List<PcapIf> alldevs = null;

    // 获取错误信息,由于库函数中的形参是StringBuilder类型的
    /*获取所有网络适配器*/
    public  static List<PcapIf> findDevices() 
    {
        // alldevs用来装载所有的network interface card，
        alldevs = new ArrayList<PcapIf>();     
        //首先我们要来获取系统中的设备列表
        int r = Pcap.findAllDevs(alldevs, errbuf);
        if (r != Pcap.OK || alldevs.isEmpty()) {  
            System.err.printf("Can't read list of devices, error is %s", 
                errbuf.toString());  
            return null;  
        }  
        return alldevs;    
    }

    /*把10进制的数字字节转化为MAC格式（16进制）,迭代取出设备列表
    *举例：50:7B:9D:1C:8D:62(50,9D)
    *50          ：80（直接打印10进制），101_0000（转化为二进制打印）
    *ffffff9d : -99                    ，111111111111111111111111_1001_1101(JVM原理造成的)
    *可见为了按照标准格式输出必须处理一下
    *字节范围8位：-128~127。
    */
    public static String[] formatStringOutDevs(List<PcapIf> alldevs ) 
    {  
        //System.out.println("Network devices found:");  
        // iterate through all the interface and get the HW addresses 
        int i = 0; 
        byte[] mac=null;//因为硬件MAC地址是6位范围（-128~127）的数字
        StringBuilder buf =null;
        String[] devsStr=new String[3];//选择可用网卡
        for (final PcapIf device : alldevs){      
            String description = (device.getDescription() != null) 
           ? device.getDescription()  : "No description available";// 如果该设备介绍，则输出介绍       
            try {
                mac = device.getHardwareAddress();  
            }
            catch (IOException e){
                System.out.println("获取硬件地址异常");  
            }
            devsStr[i++]=description;
            if (mac == null){  
                continue; // Interface doesn't have a hardware address  
            }  
            buf=new StringBuilder();
            for (byte b : mac){  
                if (buf.length() != 0){  
                    buf.append(':');  
                }
                /*官网API不知道这个算法有何作用，不加也没错
                if (b >= 0 && b < 16) 
                {  buf.append('0');  }*/
                buf.append(Integer.toHexString((b < 0) ? b + 256 : b).toUpperCase()); 
            }
            //System.out.println("#" +(i++) +device.getName()+ " = " +buf.toString() +" : " +description);     
            /*输出结果  #发现i++是先输出再自增的
            Network devices found:
            #0\Device\NPF_{DB649A21-17AD-4126-9707-D1333D8F2430} = 50:7B:9D:1C:8D:62 : Realtek Ethernet Controller
            #1\Device\NPF_{80C157C3-E185-4834-82E2-A66A5484F398} = B4:6D:83:4F:28:A6 : Microsoft
            #2\Device\NPF_{A8BA9631-741C-416A-8225-C3FB90DF10FD} = B4:6D:83:4F:28:A7 : Microsoft
            */
        }
        return devsStr;
    }  

    /*打开选中的网络适配器*/
    public static Pcap open( int id )
    {    
        // 这里在测试的时候可以根据你的可用的网卡号自主判断,我用了1     
        PcapIf device = alldevs.get(id); 
        String name=(device.getDescription() != null) ? device.getDescription() : device.getName(); 
        System.out.println("Choosing " + "["+name+"]"+ " on your behalf:");
        //打开我们选中的设备
        int snaplen = 64 * 1024; //捕获的最大的byte数 <=65535字节
        //Capture all packets, no trucation 混杂模式,不截断的捕获所有包 
        int flags = Pcap.MODE_PROMISCUOUS; 
        int timeout = 10 * 1000;           //捕获数据包能容忍的超时时间；
        Pcap pcap = Pcap.openLive(device.getName(), snaplen, flags, timeout, errbuf); 
        if (pcap == null)
        {  // 如果获取的pcap是null，则返回相关的错误信息
      System.err.printf("Error while opening device for capture: "  
          + errbuf.toString());  
      return null;  
        } 
        return pcap;
    }
    /*
     把数据包写入文件
     */
    public static void fileWriter(String str) 
    {
        FileWriter fw=null;
        try 
        {    
            fw=new FileWriter("D:\\Packet.txt",true);
            fw.write(str);
        }
        catch(IOException e){ 
            System.out.println("写入文件时发生错误"); 
        }
        finally{
            try{ fw.close(); }
            catch (IOException e){ System.out.println("写文件关闭错误"); }
        }

    }

}
