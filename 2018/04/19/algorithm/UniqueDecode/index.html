唯一可译码的判断方法

Posted on 2018-04-19 | In Java | Visitors:

y原理

将码C中所有可能的尾随后缀组成一个集合F，当且仅当集合F中,没有包含任一码字，则可判断此码C为唯一可译码。
集合F的构成方法：
首先，观察码C中最短的码字是否是其它码字的前缀，
若是，将其所有可能的尾随后缀排列出。
而这些尾随后缀又有可能是某些码字的前缀，再将这
些尾随后缀产生的新的尾随后缀列出。
然后再观察这些新的尾随后缀是否是某些码字的前缀，
再将产生的尾随后缀列出，依此下去，直到没有一个
尾随后缀是码字的前缀为止。
这样，首先获得了由最短的码字能引起的所有尾随后缀，
接着，按照上述步骤将次短码字,等等所有码字可能
产生的尾随后缀全部列出。由此得到由码C的所有可能的尾随后缀的集合F。

源代码

import java.util.*;
public class UniqueDecode
{    
    //存储尾随本次后缀和原始数据集合
    private static TreeSet<String> postfix = null;
    //存储原始数据
    private static TreeSet<String> array = null;
    //存储本次尾随后缀
    private static TreeSet<String> subPostfix = null;
    //存储将要操作的后缀
    private static TreeSet<String> subPostfixAdd = null;
    private static boolean  flag = false;
    private static int i = 0;
    public static void main(String[] agrs){
        Scanner  sc   = new Scanner(System.in);
        System.out.println("请输入码字个数");
        //while(sc.hasNext()){
            //i = 0;
            array = new TreeSet<String>();
            int n=sc.nextInt();
            String[] code = new String[n];
            System.out.println("请输入码字");
            for(int i = 0;i<n;i++){
                code[i]= sc.next();
                if(!array.add(code[i])){
                    //若码字中有重复码字则返回“是奇异码”
                    System.out.println("是奇异码");
                    return ;
                }
            }
            //postfix = new TreeSet<String>(array);
            //Iterator it = postfix.iterator();
            //while(it.hasNext()){    
            //    System.out.println("###:"+it.next());
            //}
            //第一次查找
            subPostfixAdd = new TreeSet<String>();
            seek1(array);
            //之后的查找
            while (flag){
                postfix = new TreeSet<String>(array);//清空上一次的后缀
                subPostfix = new TreeSet<String>(subPostfixAdd);
                subPostfixAdd = new TreeSet<String>();//每次清空尾随后缀
                seek2(array,subPostfix);
            }
            if(i>=2){
                System.out.println("是唯一可译码");
            }
        //}
    }
    /*开始判断码字中最短的码字是否是其它码字的前缀，然后依次判断后续码字
    */
    public static void seek1(TreeSet<String> array){
        String sHigh = null;
        for(String s:array){
            postfix = new TreeSet<String>(UniqueDecode.array);//清空上一次的后缀
            //System.out.println("s:"+s);
            String temp = s;        
            //.higher返回此 set 中严格大于给定元素的最小元素；如果不存在这样的元素，则返回 null。
            while((sHigh = array.higher(temp))!=null){//判断是否是最后一个元素，是最后就结束这个子循环
                //System.out.println("sh:"+sHigh);
                //boolean startsWith(String prefix)测试此字符串是否以指定的前缀开始。
                if(sHigh.startsWith(s)){
                    //substring(int beginIndex)返回一个新的字符串，它是此字符串的一个子字符串。
                    //System.out.println("找到了+1:");
                    String fix = sHigh.substring(s.length());
                    //System.out.println(fix);
                    subPostfixAdd.add(fix);//把后缀加到子集合subF中
                    if(!postfix.add(fix)){//把后缀加到postfix集合中
                        System.out.println("不是唯一可译码");//由于尾随后缀不可能同时出现两次以上，不然会无限循环
                        System.exit(-1);
                    }
                }
                //.last()返回此 set 中当前最后一个（最高）元素。
                temp = sHigh;
            }
        }
        flag = (subPostfixAdd.size()!=0);
        //只找了一次，如果都没有后缀，那是即时码
        System.out.println("order:"+(++i));
        if(!flag){
            System.out.println("是即时码");
        }
    }
    public static void seek2(TreeSet<String> array,TreeSet<String> subPostfix){
        String sHigh = null;
        //在subPostfix中的前缀
        for(String subF:subPostfix){
            postfix = new TreeSet<String>(UniqueDecode.array);
            String temp = subF;
            while((sHigh = array.higher(temp))!=null){    
                if(sHigh.startsWith(subF)){
                    //System.out.println("找到了+1:");
                    String fix = sHigh.substring(subF.length());
                    //System.out.println(fix);
                    subPostfixAdd.add(fix);
                    if(!postfix.add(fix)){
                        System.out.println("不是唯一可译码");
                        System.exit(-1);
                    }
                }
                temp = sHigh;
            }
        }
        String subFHigh = null;
        //在array中的前缀
        for(String s:array){
            postfix = new TreeSet<String>(UniqueDecode.array);
            String temp = s;
            while((subFHigh = subPostfix.higher(temp))!=null){
                if(subFHigh.startsWith(s)){
                    //System.out.println("找到了+1:");
                    String fix = subFHigh.substring(s.length());
                    //System.out.println(fix);
                    subPostfixAdd.add(fix);
                    if(!postfix.add(fix)){
                        System.out.println("不是唯一可译码");
                        System.exit(-1);
                    }
                }
                temp = subFHigh;
            }
        }
        flag = (subPostfixAdd.size()!=0);
        System.out.println("order:"+(++i));
    }
}

总结

  • 所有的后缀不能是一个码字，否则不是唯一可以码
