局域网文件并发传输

Posted on 2018-04-13 | In Java | Visitors:
在同一个局域网内的人可以互传文件，默认服务器是我的地址（10.18.141.254），可以设置地址发给指定的某个人

客户端程序

package niehua.client;

import java.io.BufferedWriter;
import java.io.File;
import java.io.FileInputStream;
import java.io.InputStream;
import java.io.OutputStream;
import java.io.OutputStreamWriter;
import java.net.Socket;

import javax.swing.JOptionPane;

import niehua.windows.MainWindows;

public class TcpC implements Runnable{
    public  static String addr = "10.18.141.254";//服务器地址,192.168.1.129,222.28.78.50
    private Socket cls = null;
    private String dirPath= null;
    private String dirName = null;
    private FileInputStream fr = null;
    public TcpC(String dirPath,String dirName) {
        this.dirPath = dirPath;
        this.dirName = dirName;
    }
    public  void run ()
    {
        try{//建立客户端Socket
            cls=new Socket(addr,10000);//创建一个流套接字并将其连接到指定主机上的指定端口号
            //传文件名
            //把输入流写到Socket输出流,交给服务端
            BufferedWriter bufOutFileName=
                new BufferedWriter(new OutputStreamWriter(cls.getOutputStream())); 
            bufOutFileName.write(dirName);
            bufOutFileName.newLine();
            bufOutFileName.flush();
            //传文件
            File f = new File(dirPath,dirName);
            fr = new FileInputStream(f);

            OutputStream out = cls.getOutputStream();//可接受字符和字节流，且true使println自动换行刷新

            byte[] b = new byte[1024];
            int len = 0;
            while ((len=fr.read(b))!= -1){
                out.write(b,0,len);
            }
            cls.shutdownOutput();//关闭客户端输出流，相当于返回服务端结束标记-1，否则服务端将一直等待读，且文件写没法关闭，堵塞
            //从服务端接收返回数据
            InputStream in = cls.getInputStream();
            len=in.read(b);//接收客户端回复
            //通知
            String info = "From Server:"+new String(b,0,len);
            System.out.println(info);
            //showMessageDialog(Component parentComponent,Object message, String title,int messageType) throws HeadlessException
            //很不想把界面做到业务处理代码处，因为把面板公有化感觉不安全，但为了保证上传成功能正确传达给用户，还没想到更好的方法。
            JOptionPane.showMessageDialog(MainWindows.contentPane,info, "notice", JOptionPane.INFORMATION_MESSAGE);

        }catch(Exception e) {
            System.out.println(e.toString());
            //throw new RuntimeException("客户端连接错误");
        }finally {    
            try{fr.close();cls.close(); }//结束时返回-1给服务器 
            catch(Exception e1) {System.out.println(e1.toString());}//throw new RuntimeException("连接关闭错误");
        }    
    }
}

服务器端程序

/*遇到个问题，怎样让多个用户同时传数据给我？怎样使服务器端和客户端分离。->服务端开多线程
 * 第二个，主要解决传送的文件的名字，怎样传数据之前交给服务端
 * 首先：服务器必须先开着，这时文件名不知道，因为用户还没有选择。当客户端发送文件时建立连接时，
 * 服务器与之建立连接，这时可以在客户端先把文件名发给服务器，可解决
 * 想来想去，也没找到可用的方法，传文件之前先发一个文件名数据包给服务器。
*/
package niehua.server;

import java.io.BufferedReader;
import java.io.File;
import java.io.FileOutputStream;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.io.OutputStream;
import java.net.ServerSocket;
import java.net.Socket;

class ServerThread implements Runnable{
    private Socket cls= null;
    private FileOutputStream fo =null ;
    private String file= null;
    public ServerThread(Socket cls) {
        this.cls = cls;
    }
    public void run(){
        try {
            String ip=cls.getInetAddress().getHostAddress();
            System.out.println("  ip: "+ip+" connected");
            //服务端接收客户端文件名输入流
            BufferedReader bufInFileName = 
                    new BufferedReader(new InputStreamReader(cls.getInputStream())); 
            file = "C:\\"+bufInFileName.readLine();
            //服务端接收客户端文件输入流
            InputStream  in  = cls.getInputStream();
            File f = new File(file);
            fo = new FileOutputStream(f);
            byte[] b = new byte[1024];
            int len = 0;
            while ((len=in.read(b))!= -1){    
                fo.write(b,0,len);//写到输出流
            }
            OutputStream out = cls.getOutputStream();//回复客户端
            out.write("上传成功".getBytes());
        }catch(Exception e){
            System.out.println(e.toString());
            //throw new RuntimeException("Server服务错误");
        }finally {
            try{
                fo.close();
                cls.close();

            }catch(Exception e1){
                System.out.println(e1.toString());
                //throw new RuntimeException("Server关闭错误");
            }
        }
    }
}
public class TcpS implements Runnable{
    private int port = 10000;
    private ServerSocket ss= null;
    public void run() {
        try {
            ss=new ServerSocket(port);//绑定服务器端口
            while(true) {
                //阻塞方法，想办法使客户端连接之后把文件名传给
                Socket cls = ss.accept();//将ServerSocket与Socket关联起来
                new Thread(new ServerThread(cls)).start();
            }
        }catch(Exception e){
            System.out.println(e.toString());
        }
        //ss.close();
    }

}
